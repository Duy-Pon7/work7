            CREATE TABLE BangKhachHang (
IDKhachHang INT PRIMARY KEY,	-- Primary Key: IDKhachHang là khóa chính
Ten VARCHAR(100) NOT NULL,      -- Tên khách hàng
Email VARCHAR(100) UNIQUE,      -- Email là duy nhất cho mỗi khách hàng
SDT VARCHAR(15)                 -- Số điện thoại khách hàng
);

CREATE TABLE BangHoaDon (
IDBangHoaDon INT PRIMARY KEY,          -- Primary Key: IDBangHoaDon là khóa chính
NgayDatHang DATE NOT NULL,             -- Ngày đặt hàng
SoTien DECIMAL NOT NULL,        -- Số tiền của đơn hàng
IDKhachHang1 INT,                       -- Khóa ngoại liên kết với bảng BangKhachHang
FOREIGN KEY (IDKhachHang1) REFERENCES BangKhachHang(IDKhachHang)  -- Foreign Key: BangHoaDon tham chiếu đến bảng BangKhachHang qua khóa ngoại
);
